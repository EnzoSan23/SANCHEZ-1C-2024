116void read_data(uint8_t * data, uint8_t length)
117{
118 if(data[0] == 'O')
119 {
120 encendido = true;
121 }
122
123 if(data[0] == 'o')
124 {
125 encendido = false;
126 }
127
128 if ((data[0] == 'V') & (!dosis_seteada))
129 {
130 dosisRequerida = 0;
131 i = 1;
132 while (data[i] != 'A')
133 {
134 dosisRequerida = dosisRequerida * 10;
135 dosisRequerida = dosisRequerida + (data[i] - '0');
136 i++;
137 }
138 dosis_seteada = true;
139 }

233static void pesar(void* param)
234{
235 while (true)
236 {
237 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
238 if (encendido)
239 {
240 AnalogInputReadSingle(CH0, &pesoElectronico);
241
242 dosisInstantanea = -1*(pesoElectronico - 3477)/(0.74); //Ecuacion con el recipiente tenido en cuenta
243
244 if (dosisInstantanea >= dosisRequerida)
245 {
246 dosis_seteada = false;
247 }
248 controlLEDs();
249 }
250 }

260static void controlValvular(void* param)
261{
262 while (true)
263 {
264 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
265 if (encendido)
266 {
267 if ((dosis_seteada) & (dosisInstantanea <= dosisRequerida*0.5))
268 {
269 ServoMove(SERVO_0, 180); //180 grados -> valvula abierta
270 UartSendString(UART_PC, "Valvula abierta \r\n");
271 }
272 else if ((dosis_seteada) & (dosisInstantanea > dosisRequerida*0.5) & (dosisInstantanea <= dosisRequerida))
273 {
274 ServoMove(SERVO_0, 0); //0 grados -> valvula medio cerrada
275 UartSendString(UART_PC, "Valvula medio abierta \r\n");
276 }
277 else if ((dosisInstantanea >= dosisRequerida) & (!dosis_seteada))
278 {
279 ServoMove(SERVO_0, 90); //90 grados -> valvula cerrada
280 UartSendString(UART_PC, "Valvula cerrada \r\n");
281 }
282 }
